public class SimpleAsset implements Contract {
    private final Map<String, String> assetMap = new HashMap<>();
    
    // 创建资产
    public void createAsset(Context ctx, String assetId, String value) {
        if (assetMap.containsKey(assetId)) {
            throw new RuntimeException("Asset already exists");
        }
        assetMap.put(assetId, value);
    }
    
    // 读取资产
    public String readAsset(Context ctx, String assetId) {
        String asset = assetMap.get(assetId);
        if (asset == null) {
            throw new RuntimeException("Asset not found");
        }
        return asset;
    }
}
